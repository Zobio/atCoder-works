template <class RandomAccessIterator, typename T>
RandomAccessIterator rupper_bound(RandomAccessIterator first, RandomAccessIterator last, T num) {
	//ソートされた数列の中で、num未満となる要素の内、最も添え字が最大である要素のイテレータを返す
	long long siz = last - first;
	vector<T> tmp(siz);
	for(long long i = 0; i < siz; i++) tmp[i] = -*(last - 1 - i);
	return last - 1 - (upper_bound(tmp.begin(), tmp.end(), -num) - tmp.begin()); 
}
