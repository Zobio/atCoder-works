string convert(long long x, long long base){
    /*10進数のxを(base)進数に進数変換して返す
    (例)
    convert(1000, 2) --> "1111101000"
    注意 : 2 <= base <= 10
    */
    string res;
    while(x){
        res.push_back('0' + (x % base));
        x /= base;
    }
    reverse(res.begin(),res.end());
    return res;
}
