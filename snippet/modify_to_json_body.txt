template <class T>
vector<vector<T>> mat_pow(vector<vector<T>> a, long long n) {
    // 行列累乗
    vector<vector<T>> res(a.size(), vector<long long>(a.size()));
    // 単位行列で初期化
    for (int i = 0; i < a.size(); i++) res[i][i] = 1;
	// 繰り返し二乗法
	while (n > 0) {
		if (n & 1) res = mat_mul(a, res);
		a = mat_mul(a, a);
		n >>= 1;
	}
	return res;
}
