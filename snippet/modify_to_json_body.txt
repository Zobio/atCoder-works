template<typename T>
vector<pair<T, T>> RLE(vector<T> arr) {
	/*
	ランレングス圧縮をする。
	(例) {1,1,1,2,4,4,4,4,4,3,3,3} --> {{1,3},{2,1},{4,5},{3,3}}
	*/
	vector<pair<T, T>> ret;
	ret.push_back({arr.front(), 1});
	for(ll i = 1; i < arr.size(); i++) {
		if(arr[i] == ret.back().first) ret.back().second++;
		else ret.push_back({arr[i], 1});
	}
	return ret;
}
