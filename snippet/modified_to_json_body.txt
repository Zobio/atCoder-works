"template <class T>",
"T comb(T P_, T Q_, T mo) {",
"	static const int N_=3020; //要変更",
"	assert(max(P_, Q_) < N_);",
"	static T C_[N_][N_];",
"	",
"	if(C_[0][0]==0) {",
"		for(int i = 0; i < N_; i++) {",
"			C_[i][0] = C_[i][i] = 1;",
"		}",
"		for(int i = 1; i < N_; i++) for(int j = 1; j < i; j++) {",
"			C_[i][j] = (C_[i - 1][j - 1] + C_[i - 1][j]) % mo;",
"		}",
"	}",
"	return C_[P_][Q_];",
"}"
