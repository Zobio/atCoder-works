"template <typename T>",
"vector<T> enum_primes(T num) {",
"	//num以下の素数を列挙する",
"	//O(NloglogN)",
"	vector<bool> isPrime(num + 1, true);",
"	isPrime[0] = isPrime[1] = false;",
"	for(T i = 2; i * i <= num; i++) {",
"		if(!isPrime[i]) continue; //既に操作済み",
"		for(T j = i * 2; j <= num; j += i) {",
"			isPrime[j] = false;",
"		}",
"	}",
"	vector<T> ret;",
"	for(T i = 2; i <= num; i++) {",
"		if(isPrime[i]) ret.push_back(i);",
"	}",
"	return ret;",
"}"
