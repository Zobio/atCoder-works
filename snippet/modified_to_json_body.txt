"template<typename V>",
"V shrink_coordinates(V a) {",
"    /*",
"    座標圧縮を行う関数",
"    (例)",
"    {6, 9, 9, 2, 100} -->  {1, 2, 2, 0, 3}",
"    */",
"    V b = a;",
"    sort(b.begin(), b.end());",
"    b.erase(unique(b.begin(), b.end()), b.end());",
"    V res(a.size());",
"    for(ll i = 0; i < a.size(); i++) {",
"    	res.at(i) = lower_bound(b.begin(), b.end(), a.at(i)) - b.begin();",
"    }",
"    return res;",
"}"
